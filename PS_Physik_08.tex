\section*{96. Zwei Himmelskörper}
	\begin{multicols}{3}
	\begin{flalign*}
		\vec{v}_1 &= \vector{v_1 \\ 0} &&\\
		\abs{v_1} &= v_1 &&
	\end{flalign*}
	\columnbreak
	\begin{flalign*}
		\\ \vec{v}_2 &= \vector{0 \\ v_2} &&\\
		\abs{v_2} &= v_2 &&
	\end{flalign*}
	\columnbreak 
	\begin{flalign*}
		\\ \vec{v}_E &= \tfrac{1}{m_1+m_2}\vector{m_1v_1 \\ m_2v_2} &&\\
		\abs{v_E} &= \tfrac{\sqrt{(m_1v_1)^2+(m_2v_2)^2}}{m_1+m_2} &&
	\end{flalign*}
	\end{multicols}

\begin{enumerate}
	\item \dl{\( \alpha = \arctan(\tfrac{m_2v_2}{m_1v_1}) \)} \\
	\item \( K_1 := E_{kin}\) vor dem Stoß \(;\quad K_2 := E_{kin} \) nach dem Stoß 
	\begin{flalign*}
		K_1 &= K_2 + \Delta E &&\\
		\tfrac{m_1 v_1\!^2}{2} + \tfrac{m_2 v_2\!^2}{2} &= \tfrac{1}{2} \left(m_1+m_2\right) \left( \tfrac{\sqrt{(m_1v_1)^2+(m_2v_2)^2}}{m_1+m_2}\right)^2 + \Delta E &&\\
		\Delta E &= \tfrac{(m_1v_1)^2 + (m_2v_2)^2 + m_1m_2 v_1\!^2 + m_1m_2 v_2\!^2 - (m_1v_1)^2 - (m_2v_2)^2}{2(m_1+m_2)} &&\\
		\Delta E &= \dl{\tfrac{m_1 m_2 \left(v_1\!^2+v_2\!^2\right)}{2 \left(m_1+m_2\right)}} &&\\
	\end{flalign*}
	\item 
	\begin{multicols}{2}
		Für \( m_1 = m_2 = m \) \\[1ex]
		\( \dl{\Delta E = \tfrac{1}{4} m \left(v_1\!^2 + v_2\!^2\right)} \)
		\columnbreak
		\\Für \( m_1 \gg m_2 \) \\[1ex]
		\( \dl{\Delta E = \tfrac{1}{2} m_2 \left(v_1\!^2+v_2\!^2\right)} \)
	\end{multicols}
\end{enumerate}
