\begin{mybox}{Materiewellen und Wellenfunktionen}

		\item \(  \)

			1 &= \uint[-\infty,\infty]{\abs{\tikzmark{1} \Psi(x) \tikzmark{2}}^2}{x} = \uint[0,a]{A^2x^2(a-x)^2}{x} = \tfrac{a^5A^2}{30} &&\\[2em]
			A &= \dl{\sqrt{\tfrac{30}{a^5}}} &&
		\end{flalign*}
		\AddUnderBrace[0.3]{1}{2}{\( [\text{m}^{-\tfrac{1}{2}}] \)}
	\tcbline

		\begin{flalign*}

\begin{mybox}{Neutronen im Interferometer}
	\centering \(  \)
	\tcblower
	\begin{enumerate}
		\item \( \phi_{\text{AC}} - \phi_{\text{BD}} = 0 \quad \) due to symmetry
		\begin{flalign*}
			\tfrac{p_{\text{f}}^2}{2m} &= \tfrac{p_{\text{i}}^2}{2m} - mgd &&\\
			p_{\text{f}} &= \sqrt{p_{\text{i}}^2 - 2gdm^2} = \sqrt{\tfrac{h^2}{\lambda_0^2} - 2gdm^2} &&\\
			\lambda_{\text{f}} &= \tfrac{h}{p_{\text{f}}} = \tfrac{h}{\sqrt{\tfrac{h^2}{\lambda_0^2} - 2gdm^2}} = \tfrac{1}{\sqrt{\tfrac{1}{\lambda_0^2} - \tfrac{2gdm^2}{h^2}}} &&
		\end{flalign*}
		\( \phi_{\text{AB}} - \phi_{\text{CD}} = k_{\text{i}}L - k_{\text{f}}L = 2\pi L\left(\tfrac{1}{\lambda_0} - \tfrac{1}{\lambda_{\text{f}}}\right) = \dl{2\pi L\left(\tfrac{1}{\lambda_0} - \sqrt{\tfrac{1}{\lambda_0^2} - \tfrac{2gdm^2}{h^2}} \right)} \)
	\tcbline
		\item \( \lambda_0 = 0.2 \unit{m};\quad L = 45 \unit{mm};\quad d = 24 \unit{mm};\quad \Delta\phi = \ang{81} \)
		\begin{flalign*}
			\psi_1 &= \psi_2\expo[2\Delta\phi + \pi/2][\iu] &&\\
			\psi_2 &= \tfrac{\psi_0}{\sqrt{2}} &&\\
			\psi_1' &= \tfrac{1}{\sqrt{2}}(\psi_1 + \iu\psi_2) = \tfrac{\psi_2}{\sqrt{2}}(\expo[2\Delta\phi + \pi/2][\iu] + \iu) = \tfrac{\psi_2}{\sqrt{2}}\expo[\pi/2][\iu](\expo[\Delta\phi\iu] + 1) &&\\
			\psi_2' &= \tfrac{1}{\sqrt{2}}(\psi_2 + \iu\psi_1) = \tfrac{\psi_2}{\sqrt{2}}(\iu\expo[2\Delta\phi + \pi/2][\iu] + 1) = \tfrac{\psi_2}{\sqrt{2}}(\expo[(\Delta\phi + \pi)\iu] + 1) &&\\
		\end{flalign*}
		\begin{flalign*}
			I_0 &\propto \abs{\psi_0}^2 = 5000 \unit{1/s} &&\\
			I_1 &\propto \abs{\psi_1'}^2 = \abs{\tfrac{\psi_2}{\sqrt{2}}\expo[\pi/2][\iu](\expo[\Delta\phi\iu] + 1)}^2 = \tfrac{\psi_0^2}{2}\abs{1 + \cos(\Delta\phi)} = \dl{2891\unit{1/s}} &&\\
			I_2 &\propto \abs{\tfrac{\psi_2}{\sqrt{2}}(\expo[(\Delta\phi + \pi)\iu] + 1)}^2 = \tfrac{\psi_0^2}{2}\abs{1 - \cos(\Delta\phi)} = \dl{2109 \unit{1/s}} &&
		\end{flalign*}
	\end{enumerate}
\end{mybox}
