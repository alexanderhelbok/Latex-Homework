\section*{177. Stabpendel mit Kugeln}
	\begin{align*}
		L &= 0.3 \unit{m};\ &r_1 &= 0.02 \unit{m};\ &r_2 &= 0.03 \unit{m} &&\\
		m_0 &= 0.1 \unit{kg};\ &m_1 &= 0.1 \unit{kg};\ &m_2 &= 0.3 \unit{kg} &&
	\end{align*}
	\begin{enumerate}
	\item \( I_{ges} = I_{Stab} + I_{Kugel_{1}} + I_{Kugel_{2}} \)	
		\begin{flalign*}
			I_{Stab} &= \tfrac{1}{12}m_0L^2 = 7.5 * 10^{-4} \unit{kg.m^2} &&\\
			I_{Kugel_{1}} &= \tfrac{2}{5}m_1r_1\!^2 + m_1(\tfrac{L}{2}+r_1)^2 = 2.9*10^{-3} \unit{kg.m^2} &&\\
			I_{Kugel_{2}} &= \tfrac{2}{5}m_2r_2\!^2 + m_2(\tfrac{L}{2}+r_2)^2 = 9.8*10^{-3} \unit{kg.m^2} &&\\[2ex]
			\Rightarrow I_{ges} &= \dl{1.3 * 10^{-2} \unit{kg.m^2}} &&
		\end{flalign*}
	\item 
		\begin{flalign*}
			x_{Stab_{sp}} &=  0 \unit{m} &&\\
			x_{Kugeln_{sp}} &= \tfrac{m_1 \left(-\tfrac{2r_1 + L}{2} \right) + m_2 \left(\tfrac{2r_2 + L}{2} \right)}{m_1+m_2} = 0.0925 \unit{m} &&\\[2ex]
			x_{sp} &= \tfrac{x_{Stab_{sp}}*m_0 + x_{Kugeln_{sp}}*(m_1+m_2)}{m_0+m_1+m_2} = \dl{0.074} \unit{m} &&
		\end{flalign*} 
	\item \( T = 2\pi\sqrt{\tfrac{I}{Mgl}} \)
		\begin{flalign*}
			T &= 2\pi\sqrt{\tfrac{I}{(m_0+m_1+m_2)\: g\: \left(\tfrac{2r_2+L}{2}\right)}} &&\\
			&= \dl{1.2 \unit{s}} &&
		\end{flalign*}
	\end{enumerate}

\section*{184. Resonanzverhalten einer Stimmgabel}
	\( f_0 = 440 \unit{Hz};\quad \omega_0 = 2\pi f_0;\quad x(\tau) = x_m\mathrm{e}^{-\beta \tau/2};\quad x(5) = \tfrac{1}{10}x(0) \)\\
	\begin{enumerate}
		\item \(  \)
		\begin{flalign*}
			\tfrac{x_m}{10} &= x_m\mathrm{e}^{-\beta 5/2}&&\\
			\beta &= -\tfrac{2}{5}\ln(0.1) &&\\[2ex]
			\tau &= \tfrac{1}{\beta} = \dl{1.09 \unit{s}} &&
		\end{flalign*}
		\item \( \omega' = \sqrt{\omega_0^2 - \tfrac{\beta^2}{4}} \)\\
		\begin{flalign*}
			\omega' &= 2764.6 \unit{Hz}  &(\omega_0 \approx \omega') &&\\
			f' &= \tfrac{\omega'}{2\pi} = 440 \unit{Hz}  &(f_0 \approx f') &&\\
			f'*\tau &= \dl{477.7} &&
		\end{flalign*}
		\item \( Q = \tfrac{\omega'}{\beta} \)
		\begin{flalign*}
			Q &= \dl{3001.63} &&
		\end{flalign*}
		\item \( \delta f = \tfrac{f'}{Q} \)
		\begin{flalign*}
			\delta f =&\; \dl{0.15 \unit{Hz}} &&
		\end{flalign*}
		\item \( x_m(f) = \tfrac{2\pi f_0}{\sqrt{16\pi^2(f - f_0)^2 + \beta^2}};\quad E(f) = \tfrac{(\pi f_0)^2}{4\pi^2(f-f_0)^2 + (\beta/2)^2};\quad f_1 = 439.5 \unit{Hz} \)
		\begin{flalign*}
			x_m(f_0) &= \tfrac{2\pi f_0}{\sqrt{16\pi^2(f_0 - f_0)^2 + \beta^2}} = 3001.63 \unit{m} &&\\
			x_m(f_1) &= \tfrac{2\pi f_0}{\sqrt{16\pi^2(f_1 - f_0)^2 + \beta^2}} = 435.348 \unit{m} &&\\
			x_{rel} &= \tfrac{x_m(f_1)}{x_m(f_0)} = \dl{0.15} &&\\[2ex]
			E(f_0)  &= \tfrac{(\pi f_0)^2}{4\pi^2(f_0-f_0)^2 + (\beta/2)^2} = 9.0 * 10^6 \unit{J} &&\\
			E(f_1)  &= \tfrac{(\pi f_0)^2}{4\pi^2(f_1-f_0)^2 + (\beta/2)^2} = 1.9 * 10^5 \unit{J} &&\\
			E_{rel} &= \tfrac{E(f_1)}{E(f_0)} = \dl{0.021} &&\\
		\end{flalign*}
	\end{enumerate}
